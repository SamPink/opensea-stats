AG_sum <- ape_gang_sales %>% 
  mutate(date = as_date(time)) %>% 
  group_by(date) %>% 
  summarise(date_mean = mean(sale_price),n_sales= n(),date_volume = sum(sale_price)) %>% 
  mutate(cum_volume = cumsum(date_volume))
